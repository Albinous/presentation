							if (true) {
								// scope
								const message = 'Hello World!';
								console.log(message); // 'Hello World!'
							}
							console.log(message); // ReferenceError

The scope is a policy that manages the accessibility of variables.

Scope types

  • Global
  • Block
  • Function

Global scope

  • outermost scope
  • Any variable that’s not inside any function or block, is inside the global scope.
  • accessible from any inner scope

Global scope


							let programmingLanguageName = "JavaScript";
							// code here can use programmingLanguageName

Global scope


							let programmingLanguageName = "JavaScript";
							// code here can use programmingLanguageName
							
							function myFunction() {
								// code here can also use programmingLanguageName
							}

Block Scope

  • defined by pair of {} curly braces
  • for variables declared using let and const
  • can be accessed only inside of {}

Block Scope


							for (const word of ['Rolling', 'Scopes', 'School']) {
								// "for" block scope
								const message = 'Welcome';
								console.log(word);   // 'Rolling', 'Scopes', 'School'
								console.log(message); // 'Welcome', 'Welcome', 'Welcome'
							}
							
							console.log(word);   // throws ReferenceError
							console.log(message); // throws ReferenceError

！！！ "var" is not block scoped


							if (true) {
								// "if" block scope
								var count = 0;
								console.log(count); // 0
							}

							console.log(count); // 0

Function Scope

Variables declared inside a function

Function Scope


							function run() {
								// "run" function scope
								var message = 'Run, Forrest, Run!';
								console.log(message); // 'Run, Forrest, Run!'
							}

							run();
							console.log(message); // throws ReferenceError

Scope's shadowing


							var message = 'Hello';

							function shadowing() {
								var message = 'Oh! I am changed';

								function alsoShadowing() {
									var message = 'Here another new value!';

									console.log(message); // 'Here another new value!'
								}

								alsoShadowing();
								console.log(message); // 'Oh! I am changed'
							}

							shadowing();

Variables isolation


						function foo() {
							// "foo" function scope
							let count = 0;
							console.log(count); // 0
						}
						function bar() {
							// "bar" function scope
							let count = 1;
							console.log(count); // 1
						}
						foo();
						bar();
